48 template <class T>
49 nlohmann::json internalSafeGet(nlohmann::json current, T key, bool expectNonEmpty)
50 {
51 nlohmann::json result = internalGet<T>(current, key);
52 if (result.is_null() || result.empty())
53 {
54 if (expectNonEmpty)
55 {
56 JSONSingletonErrorHandler::getInstance().handle(JSONError::fieldEmptyError, __FILE__, __LINE__);
57 throw 1;
58 }

77 template<class T> nlohmann::json* internalBranch(T key)
78 {
79 nlohmann::json result;
80 try

83 }

95 {

127 template<class O>

129 {

170 template<class T, class O>
171 O get(T key, bool expectNonEmpty = false)
172 {
173 nlohmann::json result = internalSafeGet<T>(*json, key, expectNonEmpty);
174 if (result.empty() || result.is_null())
175 {
176 return getDefault<O>();
177 }
178 O finalResult;
179 try
180 {
181 finalResult = (O)result;
182 }
183 catch (nlohmann::json::type_error)
184 {
185 JSONSingletonErrorHandler::getInstance().handle(JSONError::typeError, __FILE__, __LINE__);
186 throw 1;
187 }
188 return finalResult;
189 }

199 template <class T, class O>
200 O repeatedGet(std::vector<T> const &keys, bool expectNonEmpty = false)
201 {
202 nlohmann::json current = *json;
203 for (int i = 0; i < keys.size(); i++)
204 {
205 T key = keys[i];
206 current = internalSafeGet(current, key, expectNonEmpty);
207 if (current.empty() || current.is_null())
208 {
209 if (i == keys.size() - 1)
210 {
211 return getDefault<O>();
212 }
213 else
214 {
215 JSONSingletonErrorHandler::getInstance().handle(JSONError::fieldEmptyError, __FILE__, __LINE__);
216 throw 1;
217 }
218 }
219 }
220 O finalResult;
221 try
222 {
223 finalResult = (O)current;
224 }
225 catch (nlohmann::json::type_error)
226 {
227 JSONSingletonErrorHandler::getInstance().handle(JSONError::typeError, __FILE__, __LINE__);
228 throw 1;
229 }
230 return finalResult;
231 }
