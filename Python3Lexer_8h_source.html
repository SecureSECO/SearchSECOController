123 // Calculates the indentation of the provided spaces, taking the
124 // following rules into account:
125 //
126 // "Tabs are replaced (from left to right) by one to eight spaces
127 // such that the total number of characters up to and including
128 // the replacement is a multiple of eight [...]"
129 //
130 // -- https://docs.python.org/3.1/reference/lexical_analysis.html#indentation
131 static int getIndentationCount(const std::string& spaces) {
132 int count = 0;
133 for (char ch : spaces) {
134 switch (ch) {
135 case '\t':
136 count += 8 - (count % 8);
137 break;
138 default:
139 // A normal space char.
140 count++;
141 }
142 }
143
144 return count;
145 }
