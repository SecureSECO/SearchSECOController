63 /*
64 * The parameter specifies a type of data that should be shared. This may be set to one of the values described below.
65 * CURL_LOCK_DATA_COOKIE: Cookie data will be shared across the easy handles using this shared object.
66 * CURL_LOCK_DATA_DNS: Cached DNS hosts will be shared across the easy handles using this shared object.
67 * Note that when you use the multi interface, all easy handles added to the same multi handle will share
68 * DNS cache by default without this having to be used!
69 * CURL_LOCK_DATA_SSL_SESSION: SSL session IDs will be shared across the easy handles using this shared
70 * object. This will reduce the time spent in the SSL handshake when reconnecting to the same server.
71 * Note SSL session IDs are reused within the same easy handle by default. Note this symbol was added
72 * in 7.10.3 but was not implemented until 7.23.0.
73 */
74 CURLCPP_DEFINE_OPTION(CURLSHOPT_SHARE, int);
